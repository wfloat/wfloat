cmake_minimum_required(VERSION 3.18)

project(wfloat_rn_llm_jni LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

get_filename_component(WFLOAT_PACKAGE_DIR "${CMAKE_CURRENT_LIST_DIR}/../.." ABSOLUTE)
get_filename_component(WFLOAT_REPO_ROOT "${WFLOAT_PACKAGE_DIR}/../.." ABSOLUTE)

set(BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(WFLOAT_ENABLE_LLAMA_CPP ON CACHE BOOL "" FORCE)
set(WFLOAT_CORE_ENABLE_SPEECH OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_COMMON OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_TOOLS OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_SERVER OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_APP OFF CACHE BOOL "" FORCE)
set(LLAMA_BUILD_UI OFF CACHE BOOL "" FORCE)
set(LLAMA_OPENSSL OFF CACHE BOOL "" FORCE)
set(GGML_METAL OFF CACHE BOOL "" FORCE)
set(GGML_VULKAN OFF CACHE BOOL "" FORCE)
set(GGML_OPENCL OFF CACHE BOOL "" FORCE)
set(GGML_LLAMAFILE OFF CACHE BOOL "" FORCE)
set(GGML_NATIVE OFF CACHE BOOL "" FORCE)
set(GGML_OPENMP OFF CACHE BOOL "" FORCE)
set(ANDROID_SUPPORT_FLEXIBLE_PAGE_SIZES ON CACHE BOOL "" FORCE)

if(ANDROID_ABI MATCHES "arm64-v8a|armeabi-v7a")
  set(GGML_NEON ON CACHE BOOL "" FORCE)
else()
  set(GGML_NEON OFF CACHE BOOL "" FORCE)
endif()

if(ANDROID_ABI STREQUAL "arm64-v8a")
  set(GGML_CPU_KLEIDIAI ON CACHE BOOL "" FORCE)
else()
  set(GGML_CPU_KLEIDIAI OFF CACHE BOOL "" FORCE)
endif()

add_subdirectory(
  "${WFLOAT_REPO_ROOT}/vendor/llama.cpp"
  "${CMAKE_CURRENT_BINARY_DIR}/llama.cpp"
)

add_subdirectory(
  "${WFLOAT_REPO_ROOT}/native/wfloat-core"
  "${CMAKE_CURRENT_BINARY_DIR}/wfloat-core"
)

add_library(wfloat-llm-jni SHARED
  wfloat_llm_jni.cpp
)

target_include_directories(wfloat-llm-jni
  PRIVATE
    "${WFLOAT_REPO_ROOT}/native/wfloat-core/include"
)

target_link_libraries(wfloat-llm-jni
  PRIVATE
    wfloat-core
    log
)
